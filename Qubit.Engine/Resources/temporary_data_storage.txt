struct vs_in
{
    float3 position : POSITION;
    float3 color : COLOR;
};

struct vs_out
{
    float4 position_clip : SV_POSITION;
    float3 outColor : COLOR;
};

vs_out vs_main(vs_in input)
{
    vs_out output;
    
    // Just pass position directly without transformations
    output.position_clip = float4(input.position.xy, 0.5, 1.0);
    
    // Pass the color
    output.outColor = input.color;
    
    return output;
}
